const STORAGE_KEY = "srs_planner_v2";
const save = (data) => {
  try { window.storage?.set(STORAGE_KEY, JSON.stringify(data), false); } catch {}

// √¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨ Schedule engine √¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨
function buildSchedule(totalDays, inputs, startDay) {
  let lecNum = 1;
  const studyDays = [];

  for (let d = 1; d <= totalDays; d++) {
    const raw = inputs[d];
    const count = raw === undefined ? 2 : Math.max(0, parseInt(raw) || 0);
    const lecs = [];
    for (let i = 0; i < count; i++) lecs.push(lecNum++);
    studyDays.push({ day: d, count, lecs });
  }

  const totalLectures = lecNum - 1;
  const planEnd = totalDays + 90;

  const revMap = {};
  for (let d = 1; d <= planEnd; d++) revMap[d] = [];

  for (const { day, lecs } of studyDays) {
    for (const ln of lecs) {
      for (let i = 0; i < INTERVALS.length; i++) {
        const rd = day + INTERVALS[i];
        if (rd <= planEnd) revMap[rd].push({ ln, iv: INTERVALS[i], ivLabel: LABELS[i], fromDay: day });
      }
    }
  }

  const allDays = [];
  for (let d = 1; d <= planEnd; d++) {
    const sd = studyDays[d - 1];
    allDays.push({
      day: d,
      isStudy: d <= totalDays,
      count: sd?.count ?? 0,
      lecs: sd?.lecs ?? [],
      revisions: revMap[d],
    });
  }

  // "today" based on startDay offset
  const todayRow = startDay >= 1 && startDay <= planEnd ? allDays[startDay - 1] : null;

  return { allDays, totalLectures, planEnd, studyDays, todayRow };
}

const toYMD = (date) => date.toISOString().slice(0, 10); // "YYYY-MM-DD"
const todayYMD = () => toYMD(new Date());

function dateForDay(startDateStr, dayNum) {
  // dayNum is 1-based; Day 1 = startDate
  if (!startDateStr) return null;
  const d = new Date(startDateStr);
  d.setDate(d.getDate() + dayNum - 1);
  return d;
}

function formatDate(date) {
  if (!date) return "";
  return date.toLocaleDateString("en-IN", { day: "numeric", month: "short", year: "numeric", weekday: "short" });
}

function formatDateShort(date) {
  if (!date) return "";
  return date.toLocaleDateString("en-IN", { day: "numeric", month: "short", weekday: "short" });
}

// Given a startDate and today's real date, figure out which Day number it is
function currentDayNum(startDateStr) {
  if (!startDateStr) return 1;
  const start = new Date(startDateStr);
  const now = new Date();
  start.setHours(0, 0, 0, 0);
  now.setHours(0, 0, 0, 0);
  const diff = Math.round((now - start) / 86400000) + 1;
  return Math.max(1, diff);
}

            ) : sched.todayRow.count === 0 && sched.todayRow.isStudy ? (
              <Callout color={C.red}>√∞≈∏≈°¬´ Holiday √¢‚Ç¨‚Äù No lectures or revisions today. Rest up!</Callout>
            ) : (
              <div style={{ display: "flex", flexDirection: "column", gap: 12 }}>

                {/* New Lectures */}
                {sched.todayRow.lecs.length > 0 && (
                  <Section title="√∞≈∏‚Äú‚Äî New Lectures Today" color={C.green}>
                    {sched.todayRow.lecs.map(ln => {
                      const key = `new-${ln}`;
                      const isDone = !!done[key];
                      return (
                        <TaskRow key={ln} done={isDone} color={C.green}
                          onToggle={() => toggleDone(key)}
                          label={getLecName(ln)}
                          sublabel="Learn + same-day review"
                          onRename={() => setEditingName(ln)}
                        />
                      );
                    })}
                  </Section>
                )}

                {/* Revisions */}
                {sched.todayRow.revisions.length > 0 && (
                  <Section title="√∞≈∏‚Äù¬Å Revisions Due Today" color={C.yellow}>
                    {sched.todayRow.revisions.map((r, i) => {
                      const key = `rev-${sched.todayRow.day}-${r.ln}-${r.iv}`;
                      const isDone = !!done[key];
                      return (
                        <TaskRow key={i} done={isDone} color={C.yellow}
                          onToggle={() => toggleDone(key)}
                          label={getLecName(r.ln)}
                          sublabel={`${r.ivLabel} revision √Ç¬∑ Originally from Day ${r.fromDay}`}
                          onRename={() => setEditingName(r.ln)}
                        />
                      );
                    })}
                  </Section>
                )}

                {sched.todayRow.lecs.length === 0 && sched.todayRow.revisions.length === 0 && (
                  <Callout color={C.muted}>Nothing scheduled for Day {startDay}.</Callout>
                )}
              </div>
            )}

            {/* Rename modal */}
            {editingName !== null && (
              <div style={{ position: "fixed", inset: 0, background: "#000a", display: "flex", alignItems: "center", justifyContent: "center", zIndex: 100 }}>
                <div style={{ background: C.card, border: `1px solid ${C.border}`, borderRadius: 14, padding: 24, width: 320 }}>
                  <div style={{ fontWeight: 700, marginBottom: 12 }}>Rename Lecture {editingName}</div>
                  <input autoFocus
                    defaultValue={lectureNames[editingName] || ""}
                    onKeyDown={e => {
                      if (e.key === "Enter") {
                        setLectureNames(prev => ({ ...prev, [editingName]: e.target.value }));
                        setEditingName(null);
                        showToast("Name saved √¢≈ì‚Äú");
                      }
                      if (e.key === "Escape") setEditingName(null);
                    }}
                    placeholder={`e.g. "Chapter 3 √¢‚Ç¨‚Äú Thermodynamics"`}
                    style={{ width: "100%", ...inputStyle, marginBottom: 12, boxSizing: "border-box" }}
                  />
                  <div style={{ fontSize: 12, color: C.muted }}>Press Enter to save, Esc to cancel</div>
                </div>
              </div>
            )}
          </div>
        )}

        {/* √¢‚Äù‚Ç¨√¢‚Äù‚Ç¨ INPUT VIEW √¢‚Äù‚Ç¨√¢‚Äù‚Ç¨ */}
        {view === "input" && (
          <div>
            <div style={{ marginBottom: 16, display: "flex", alignItems: "flex-end", gap: 16, flexWrap: "wrap" }}>
              <div>
                <div style={{ fontSize: 12, color: C.muted, marginBottom: 4 }}>Total study days</div>
                <input type="number" min={1} max={365} value={totalDays}
                  onChange={e => setTotalDays(Math.max(1, Math.min(365, +e.target.value || 1)))}
                  style={{ width: 80, ...inputStyle }} />
              </div>
              <button onClick={() => { setInputs({}); showToast("Reset to 2 lectures/day"); }} style={btnStyle(C.surface)}>√¢‚ÄÝ¬∫ Reset all to 2</button>
              <button onClick={() => { const o = {}; for (let d = 1; d <= totalDays; d++) o[d] = 1; setInputs(o); showToast("Set all to 1"); }} style={btnStyle(C.surface)}>Set all to 1</button>
              <button onClick={() => { const o = {}; for (let d = 1; d <= totalDays; d++) o[d] = 0; setInputs(o); showToast("Set all to holiday"); }} style={btnStyle(C.redDim)}>Set all to holiday</button>
            </div>

            <div style={{ fontSize: 13, color: C.muted, marginBottom: 14 }}>
              Type lectures per day. <b style={{ color: C.text }}>0 = holiday √Ç¬∑ 1, 2, 3√¢‚Ç¨¬¶ = lectures that day.</b> Lecture numbers auto-assign based on order.
            </div>

            {/* Running lecture counter preview */}
            <div style={{ background: C.surface, border: `1px solid ${C.border}`, borderRadius: 10, padding: "10px 14px", marginBottom: 16, fontSize: 13 }}>
              <span style={{ color: C.muted }}>Lectures assigned so far: </span>
              <span style={{ color: C.accent, fontWeight: 700 }}>L1 √¢‚ÄÝ‚Äô L{sched.totalLectures}</span>
              <span style={{ color: C.muted }}> across {totalDays} days</span>
            </div>

            <div style={{ display: "grid", gridTemplateColumns: "repeat(auto-fill, minmax(100px, 1fr))", gap: 8 }}>
              {Array.from({ length: totalDays }, (_, i) => i + 1).map(day => {
                const raw = inputs[day] ?? "2";
                const num = parseInt(raw) || 0;
                const isHoliday = num === 0;
                const color = isHoliday ? C.red : num === 1 ? C.yellow : num >= 3 ? C.green : C.accent;
                const bg = isHoliday ? C.redDim : num === 1 ? C.yellowDim : num >= 3 ? C.greenDim : C.accentDim;
                // Find which lectures this day gets
                const dayData = sched.studyDays[day - 1];
                const lecRange = dayData?.lecs.length
                  ? `L${dayData.lecs[0]}${dayData.lecs.length > 1 ? `√¢‚Ç¨‚ÄúL${dayData.lecs[dayData.lecs.length - 1]}` : ""}`
                  : "";

                return (
                  <div key={day} style={{ background: bg + "88", border: `1.5px solid ${color}44`, borderRadius: 10, padding: "8px 6px", textAlign: "center" }}>
                    <div style={{ fontSize: 10, color: C.muted, marginBottom: 2 }}>Day {day}</div>
                    <input type="number" min={0} max={10} value={raw}
                      onChange={e => setInputs(prev => ({ ...prev, [day]: e.target.value }))}
                      style={{ width: 52, padding: "5px 4px", background: C.bg, border: `1px solid ${color}`, borderRadius: 6, color, fontSize: 18, fontWeight: 700, textAlign: "center" }}
                    />
                    <div style={{ fontSize: 9, color, marginTop: 3 }}>
                      {isHoliday ? "√∞≈∏≈°¬´ holiday" : `${num} lec${num !== 1 ? "s" : ""}`}
                    </div>
                    {lecRange && <div style={{ fontSize: 9, color: C.muted, marginTop: 1 }}>{lecRange}</div>}
                  </div>
                );
              })}
            </div>
          </div>
        )}

        {/* √¢‚Äù‚Ç¨√¢‚Äù‚Ç¨ SCHEDULE VIEW √¢‚Äù‚Ç¨√¢‚Äù‚Ç¨ */}
        {view === "schedule" && (
          <div>
            <div style={{ display: "flex", gap: 10, marginBottom: 14, flexWrap: "wrap", alignItems: "center" }}>
              <input placeholder="√∞≈∏‚Äù¬ç Search lecture√¢‚Ç¨¬¶" value={searchQ}
                onChange={e => setSearchQ(e.target.value)}
                style={{ flex: 1, minWidth: 160, ...inputStyle }} />
              <div style={{ display: "flex", gap: 6 }}>
                {[["√∞≈∏≈∏¬¢ New", C.green], ["√∞≈∏≈∏¬° Revision", C.yellow], ["√∞≈∏‚Äù¬¥ Holiday", C.red], ["√¢¬¨‚Ä∫ Rev-only day", C.muted]].map(([l, c]) => (
                  <span key={l} style={{ fontSize: 10, color: c, background: C.surface, padding: "3px 7px", borderRadius: 5, border: `1px solid ${c}33` }}>{l}</span>
                ))}
              </div>
            </div>

            <div style={{ display: "flex", flexDirection: "column", gap: 5 }}>
              {sched.allDays.map(row => {
                const isHoliday = row.isStudy && row.count === 0;
                const isToday = row.day === startDay;

                // Filter by search
                if (searchQ) {
                  const q = searchQ.toLowerCase();
                  const matchNew = row.lecs.some(n => getLecName(n).toLowerCase().includes(q) || `l${n}`.includes(q));
                  const matchRev = row.revisions.some(r => getLecName(r.ln).toLowerCase().includes(q) || `l${r.ln}`.includes(q));
                  if (!matchNew && !matchRev && !`day ${row.day}`.includes(q)) return null;
                }

                const borderCol = isToday ? C.accent : isHoliday ? C.red : row.lecs.length > 0 ? C.blueDim : C.border;
                const bgCol = isToday ? C.accentDim + "55" : isHoliday ? C.redDim : row.isStudy ? C.surface : C.card;

                const revGroups = INTERVALS.map((iv, idx) => ({
                  label: LABELS[idx], items: row.revisions.filter(r => r.iv === iv)
                })).filter(g => g.items.length > 0);

                return (
                  <div key={row.day} style={{ background: bgCol, border: `1px solid ${borderCol}`, borderRadius: 10, padding: "9px 14px", display: "flex", gap: 10, flexWrap: "wrap", alignItems: "center" }}>
                    <div style={{ minWidth: 80, textAlign: "center" }}>
                      <div style={{ fontSize: 9, color: C.muted }}>DAY</div>
                      <div style={{ fontSize: 19, fontWeight: 800, color: isToday ? C.accent : isHoliday ? C.red : row.isStudy ? C.blue : C.muted }}>
                        {row.day}
                      </div>
                      {startDate && (
                        <div style={{ fontSize: 9, color: isToday ? C.accent : C.muted, marginTop: 1, lineHeight: 1.3 }}>
                          {formatDateShort(dateForDay(startDate, row.day))}
                        </div>
                      )}
                      {isToday && <div style={{ fontSize: 8, color: C.accent, fontWeight: 700, marginTop: 1 }}>TODAY</div>}
                    </div>

                    <div style={{ minWidth: 110 }}>
                      <div style={{ fontSize: 9, color: C.muted, marginBottom: 2 }}>NEW</div>
                      {isHoliday ? <Tag color={C.red}>√∞≈∏≈°¬´ Holiday</Tag>
                        : row.lecs.length > 0 ? row.lecs.map(n => <Tag key={n} color={C.green}>{getLecName(n)}</Tag>)
                        : <span style={{ color: C.muted, fontSize: 11 }}>√¢‚Ç¨‚Äù</span>}
                    </div>

                    <div style={{ flex: 1, display: "flex", gap: 5, flexWrap: "wrap" }}>
                      {revGroups.map(({ label, items }) => (
                        <div key={label} style={{ background: C.yellowDim, border: `1px solid ${C.yellow}33`, borderRadius: 6, padding: "3px 9px", fontSize: 11 }}>
                          <span style={{ color: C.yellow, fontWeight: 600 }}>{label}: </span>
                          <span style={{ color: C.text }}>{items.map(r => getLecName(r.ln)).join(", ")}</span>
                        </div>
                      ))}
                      {revGroups.length === 0 && !isHoliday && row.lecs.length === 0 &&
                        <span style={{ color: C.faint, fontSize: 11 }}>No tasks</span>}
                    </div>

                    {row.revisions.length > 0 && (
                      <div style={{ textAlign: "center", minWidth: 30 }}>
                        <div style={{ fontSize: 9, color: C.muted }}>revs</div>
                        <div style={{ fontSize: 15, fontWeight: 700, color: C.yellow }}>{row.revisions.length}</div>
                      </div>
                    )}
                  </div>
                );
              })}
            </div>
          </div>
        )}

        {/* √¢‚Äù‚Ç¨√¢‚Äù‚Ç¨ SETTINGS VIEW √¢‚Äù‚Ç¨√¢‚Äù‚Ç¨ */}
        {view === "settings" && (
          <div style={{ display: "flex", flexDirection: "column", gap: 14, maxWidth: 500 }}>
            <div style={{ fontSize: 16, fontWeight: 700 }}>√¢≈°‚Ñ¢√Ø¬∏¬è Settings</div>

            {/* Start Date */}
            <SettingCard title="√∞≈∏‚Äú‚Ä¶ Start Date" desc="Set the real calendar date when Day 1 of your course began. All other days will get their dates automatically.">
              <div style={{ display: "flex", gap: 10, alignItems: "center", flexWrap: "wrap" }}>
                <input type="date" value={startDate}
                  onChange={e => setStartDate(e.target.value)}
                  style={{ ...inputStyle, width: 150 }} />
                <button onClick={() => { setStartDate(todayYMD()); showToast("Start date set to today √¢≈ì‚Äú"); }}
                  style={btnStyle(C.accent, true)}>Set to Today</button>
              </div>
              {startDate && (
                <div style={{ fontSize: 12, color: C.muted, marginTop: 8 }}>
                  Day 1 = <b style={{ color: C.text }}>{formatDate(new Date(startDate))}</b>
                  <br />Today = <b style={{ color: C.accent }}>Day {currentDayNum(startDate)}</b> √Ç¬∑ Plan ends on <b style={{ color: C.text }}>{formatDate(dateForDay(startDate, sched.planEnd))}</b>
                </div>
              )}
            </SettingCard>

            {/* Reminder */}
            <SettingCard title="√∞≈∏‚Äù‚Äù Daily Reminder" desc="Get a browser notification with today's revision list.">
              <div style={{ display: "flex", gap: 10, alignItems: "center", flexWrap: "wrap" }}>
                <input type="time" value={notifTime}
                  onChange={e => setNotifTime(e.target.value)}
                  style={{ ...inputStyle, width: 110 }} />
                {notifEnabled
                  ? <button onClick={() => { setNotifEnabled(false); showToast("Reminders off"); }} style={btnStyle(C.redDim)}>Turn Off</button>
                  : <button onClick={requestNotif} style={btnStyle(C.accent, true)}>Enable Reminder</button>
                }
                {notifEnabled && <span style={{ fontSize: 12, color: C.green }}>√¢≈ì‚Äú Active</span>}
              </div>
              <div style={{ fontSize: 11, color: C.muted, marginTop: 6 }}>
                Sends a notification at the chosen time each day. Keep this tab open for it to work.
              </div>
            </SettingCard>

            {/* Rename lectures */}
            <SettingCard title="√∞≈∏¬è¬∑√Ø¬∏¬è Name Your Lectures" desc="Give each lecture a meaningful name (e.g. 'Chapter 3 √¢‚Ç¨‚Äú Thermodynamics').">
              <div style={{ display: "flex", flexDirection: "column", gap: 6, maxHeight: 280, overflowY: "auto" }}>
                {Array.from({ length: sched.totalLectures }, (_, i) => i + 1).map(ln => (
                  <div key={ln} style={{ display: "flex", gap: 8, alignItems: "center" }}>
                    <span style={{ fontSize: 11, color: C.muted, minWidth: 20 }}>L{ln}</span>
                    <input
                      placeholder={`Lecture ${ln}`}
                      value={lectureNames[ln] || ""}
                      onChange={e => setLectureNames(prev => ({ ...prev, [ln]: e.target.value }))}
                      style={{ flex: 1, ...inputStyle, fontSize: 12, padding: "5px 8px" }}
                    />
                  </div>
                ))}
              </div>
              <button onClick={() => showToast("Names saved √¢≈ì‚Äú")} style={{ ...btnStyle(C.accent, true), marginTop: 8 }}>Save Names</button>
            </SettingCard>

            {/* Reset */}
            <SettingCard title="√∞≈∏‚Äî‚Äò√Ø¬∏¬è Reset Data" desc="Clear all progress, names, and lecture counts.">
              <button onClick={() => {
                if (confirm("Reset everything?")) {
                  setInputs({}); setDone({}); setLectureNames({}); setTotalDays(20); setStartDay(1);
                  showToast("Reset done", C.red);
                }
              }} style={btnStyle(C.redDim)}>Reset Everything</button>
            </SettingCard>

            {/* Interval info */}
            <SettingCard title="√∞≈∏‚Äú¬ê Revision Intervals" desc="The spaced repetition schedule used for every lecture.">
              <div style={{ display: "flex", flexWrap: "wrap", gap: 6 }}>
                {LABELS.map((l, i) => (
                  <span key={l} style={{ background: C.yellowDim, border: `1px solid ${C.yellow}33`, borderRadius: 6, padding: "3px 10px", fontSize: 12, color: C.yellow }}>{l}</span>
                ))}
              </div>
              <div style={{ fontSize: 11, color: C.muted, marginTop: 8 }}>8 revision sessions per lecture = permanent long-term memory.</div>
            </SettingCard>
          </div>
        )}

      </div>
    </div>
  );
}

// √¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨ Reusable sub-components √¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨√¢‚Äù‚Ç¨

function Section({ title, color, children }) {
  return (
    <div style={{ background: C.surface, border: `1px solid ${color}33`, borderRadius: 12, overflow: "hidden" }}>
      <div style={{ background: color + "18", padding: "9px 14px", fontSize: 13, fontWeight: 700, color, borderBottom: `1px solid ${color}22` }}>{title}</div>
      <div style={{ padding: "8px 10px", display: "flex", flexDirection: "column", gap: 6 }}>{children}</div>
    </div>
  );
}

function TaskRow({ done, color, onToggle, label, sublabel, onRename }) {
  return (
    <div style={{ display: "flex", alignItems: "center", gap: 10, padding: "7px 10px", background: done ? C.bg : C.card, borderRadius: 8, border: `1px solid ${done ? C.border : color + "33"}`, opacity: done ? 0.5 : 1, transition: "all 0.2s" }}>
      <button onClick={onToggle} style={{ width: 22, height: 22, borderRadius: 6, border: `2px solid ${color}`, background: done ? color : "transparent", cursor: "pointer", flexShrink: 0, display: "flex", alignItems: "center", justifyContent: "center", color: "#000", fontWeight: 900, fontSize: 13 }}>
        {done ? "√¢≈ì‚Äú" : ""}
      </button>
      <div style={{ flex: 1 }}>
        <div style={{ fontSize: 13, fontWeight: 600, textDecoration: done ? "line-through" : "none", color: done ? C.muted : C.text }}>{label}</div>
        <div style={{ fontSize: 11, color: C.muted }}>{sublabel}</div>
      </div>
      <button onClick={onRename} title="Rename" style={{ background: "none", border: "none", color: C.muted, cursor: "pointer", fontSize: 14, padding: "2px 6px", borderRadius: 5 }}>√¢≈ì¬è√Ø¬∏¬è</button>
    </div>
  );
}

function Tag({ color, children }) {
  return (
    <span style={{ display: "inline-block", background: color + "18", border: `1px solid ${color}44`, borderRadius: 5, padding: "2px 7px", fontSize: 11, color, fontWeight: 600, marginRight: 3, marginBottom: 2 }}>
      {children}
    </span>
  );
}

function Callout({ color, children }) {
  return (
    <div style={{ background: color + "18", border: `1px solid ${color}44`, borderRadius: 10, padding: "14px 18px", color, fontSize: 14 }}>
      {children}
    </div>
  );
}

function SettingCard({ title, desc, children }) {
  return (
    <div style={{ background: C.surface, border: `1px solid ${C.border}`, borderRadius: 12, padding: "14px 16px" }}>
      <div style={{ fontWeight: 700, fontSize: 14, marginBottom: 3 }}>{title}</div>
      <div style={{ fontSize: 12, color: C.muted, marginBottom: 10 }}>{desc}</div>
      {children}
    </div>
  );
}

function btnStyle(bg, filled) {
  return {
    padding: "7px 14px", borderRadius: 8, border: `1px solid ${bg}`,
    background: filled ? bg : "transparent",
    color: filled ? "#fff" : C.text,
    cursor: "pointer", fontSize: 13, fontWeight: 600,
  };
}

const inputStyle = {
  padding: "7px 10px", background: C.surface,
  border: `1px solid ${C.border}`, borderRadius: 8,
  color: C.text, fontSize: 13, outline: "none",
};
